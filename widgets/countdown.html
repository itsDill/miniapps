<!-- iowidgets Countdown Timer Widget -->
<div id="iowidgets-countdown"></div>
<script>
(function() {
    // ‚öôÔ∏è CUSTOMIZE THIS: Set your target date (YYYY-MM-DD)
    const targetDate = new Date('2025-12-31').getTime();
    
    const widget = document.getElementById('iowidgets-countdown');
    widget.style.cssText = 'background:linear-gradient(135deg,#667eea,#764ba2);border-radius:15px;padding:2rem;color:#fff;text-align:center;box-shadow:0 10px 40px rgba(0,0,0,0.3);max-width:600px;margin:0 auto;';
    
    function updateCountdown() {
        const now = new Date().getTime();
        const distance = targetDate - now;
        
        if (distance < 0) {
            widget.innerHTML = '<div style="font-size:2rem;">üéâ Event Started!</div>';
            return;
        }
        
        const days = Math.floor(distance / (1000 * 60 * 60 * 24));
        const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
        const seconds = Math.floor((distance % (1000 * 60)) / 1000);
        
        widget.innerHTML = \`
            <div style="font-size:1.5rem;margin-bottom:1.5rem;opacity:0.9;">Countdown Timer</div>
            <div style="display:grid;grid-template-columns:repeat(4,1fr);gap:1rem;">
                <div>
                    <div style="font-size:2.5rem;font-weight:bold;">${days}</div>
                    <div style="font-size:0.9rem;opacity:0.8;">Days</div>
                </div>
                <div>
                    <div style="font-size:2.5rem;font-weight:bold;">${hours}</div>
                    <div style="font-size:0.9rem;opacity:0.8;">Hours</div>
                </div>
                <div>
                    <div style="font-size:2.5rem;font-weight:bold;">${minutes}</div>
                    <div style="font-size:0.9rem;opacity:0.8;">Minutes</div>
                </div>
                <div>
                    <div style="font-size:2.5rem;font-weight:bold;">${seconds}</div>
                    <div style="font-size:0.9rem;opacity:0.8;">Seconds</div>
                </div>
            </div>
        \`;
    }
    
    updateCountdown();
    setInterval(updateCountdown, 1000);
})();
</script>
